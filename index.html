1 - Arrow Function | shorten code

          function test(X, Y) {
            return X + Y;
          }

          TO:

          let Arrow = (X, Y) => X + Y;
      

      2- var to => let & const

          var   | Global scope ,
                  redeclared,
                  When call before declare => return undefined
                  Editing
                  var test; 

          let   | locat scope,
                  NOT redeclared,
                  When call before declare => return error
                  Edit
                  let test; 

          const | locat scope,
                  NOT redeclared,
                  When call before declare => return error
                  NOT Edit
                  const test; ERROR
      

      3- Template Literals 

          let name = "mahmod"

          let test = "hi " + "im " name + "i'm coder"

          =>
          let test = `hi im ${name} i'm coder`
      

      4- Spread Operator 

          let test = [1,2,3]

          sum(test) => 123UndefinedUndefined , Use apply()
          
          sum(...test) =>  6
